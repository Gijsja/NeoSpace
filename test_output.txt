    <script>
        function getCsrfToken() {
            return document.querySelector('meta[name="csrf-token"]').getAttribute('content');
        }

        function loginForm() {
            return {
                username: '',
                password: '',
                error: '',
                loading: false,
                progressValue: 0,

                async submit() {
                    this.loading = true;
                    this.error = '';
                    this.progressValue = 0;

                    let interval = setInterval(() => {
                        if (this.progressValue < 90) this.progressValue += 5;
                    }, 50);

                    try {
                        const res = await fetch('/auth/login', {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json',
                                'X-CSRFToken': getCsrfToken()
                            },
                            body: JSON.stringify({
                                username: this.username,
                                password: this.password
                            })
                        });
                        const data = await res.json();

                        if (res.ok && data.ok) {
                            this.progressValue = 100;
                            // Trigger Cat
                            this.$dispatch('login-success');

                            setTimeout(() => {
                                window.location.href = data.redirect;
                            }, 2500); // Wait for cat to speak
                        } else {
                            // Trigger Cat
                            this.$dispatch('login-failure');

                            this.error = data.error || 'Authentication failed';
                            this.$dispatch('toast', {
                                title: 'Access Denied',
                                message: this.error,
                                type: 'error'
                            });
                        }
                    } catch (err) {
                        this.error = 'Network error: ' + err.message;
                        this.$dispatch('toast', {
                            title: 'System Failure',
                            message: this.error,
                            type: 'error'
                        });
                    } finally {
                        clearInterval(interval);
                        this.loading = false;
                    }
                }
            }
        }

        function registerForm() {
            return {
                username: '',
                password: '',
                confirmPassword: '',
                error: '',
                loading: false,
                progressValue: 0,

                async submit() {
                    if (this.password !== this.confirmPassword) {
                        this.error = "Passwords do not match.";
                        return;
                    }

                    this.loading = true;
                    this.error = '';
                    this.progressValue = 0;

                    let interval = setInterval(() => {
                        if (this.progressValue < 90) this.progressValue += 5;
                    }, 50);

                    try {
                        const res = await fetch('/auth/register', {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json',
                                'X-CSRFToken': getCsrfToken()
                            },
                            body: JSON.stringify({
                                username: this.username,
                                password: this.password
                            })
                        });
                        const data = await res.json();

                        if (res.ok && data.ok) {
                            this.progressValue = 100;
                            // Trigger cat (using login success for now as welcome)
                            this.$dispatch('login-success');

                            this.$dispatch('toast', {
                                title: 'Identity Created',
                                message: 'Welcome to the grid, ' + this.username,
                                type: 'success'
                            });
                            setTimeout(() => {
                                window.location.href = data.redirect;
                            }, 2500);
                        } else {
                            // Trigger Cat
                            this.$dispatch('login-failure');

                            this.error = data.error || 'Registration failed';
                            this.$dispatch('toast', {
                                title: 'Registration Error',
                                message: this.error,
                                type: 'error'
                            });
                        }
                    } catch (err) {
                        this.error = 'Network error: ' + err.message;
                    } finally {
                        clearInterval(interval);
                        this.loading = false;
                    }
                }
            }
        }
